Get Started

# Install the library
pip install ndp-ep

# Connect to NDP-EP
from ndp_ep import Client
client = Client(base_url="https://your-api.com", token="your-token")

# Search for data
results = client.search_datasets(terms=["satellite", "utah"])

# Download processed data
data = client.download_object("bucket", "file.nc")
